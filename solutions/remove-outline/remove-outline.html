  2. Firefox按钮 
                /* for Firefox */
            input[type=submit]::-moz-focus-inner,input[type=button]::-moz-focus-inner{    
              border : 0px; 
            }
            /* for IE8 */
            input[type=submit]:focus, input[type=button]:focus{      
                outline : none;  
            }
